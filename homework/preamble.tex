\newcommand{\makecoverpage}{
\maketitle

\newpage

% put a blank page here for hard copy version
\ifdefined\hardcopy
% for double side printing, leave backside of cover page blank

  \vspace*{\fill}
  \LARGE{\textit{This page intentionally left blank.}}
  \vspace*{\fill}
\end{center}
\thispagestyle{empty}
\else
\fi

% reset page number counter
\setcounter{page}{1}
}
